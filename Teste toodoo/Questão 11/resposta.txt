function multiplicacao (num1,num2) {
    let resultado = num1 * num2 
    return resultado
}



 multiplicacao(4,7) //  tendo como resultado  a multiplicação de 4x7 tendo o resultado de 28

 multiplicacao(5,6) //  tendo como resultado  a multiplicação de 5x6 tendo o resultado de 30
 multiplicacao(3,3) //  tendo como resultado  a multiplicação de 3x3 tendo o resultado de 9
